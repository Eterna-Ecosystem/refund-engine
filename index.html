const express = require('express'); const app = express(); const PORT = process.env.PORT || 3000; const path = require('path'); // Health check endpoint app.get('/healthz', (req, res) => { res.send('Refund Engine is running ✅'); }); // Serve static files app.use(express.static(path.join(__dirname))); // Root route → serve index.html app.get('/', (req, res) => { res.sendFile(path.join(__dirname, 'index.html')); }); app.listen(PORT, () => { console.log(`Refund Engine started on port ${PORT}`); });
